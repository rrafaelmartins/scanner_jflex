// Exemplo CORRETO de MiniJava com classe auxiliar

class Main {
    public static void main(String[] args) {
        System.out.println(new Calc().compute(10));
    }
}

class Calc {
    public int compute(int n) {
        int result;
        if (n < 20) {
            result = n + 1;
        } else {
            result = n - 1;
        }
        return result;
    }
}
